const MAX = 100, MIN = 1;
var cnt, tot, tmp;

procedure sum;
    var i;
    begin
        i := 0;
        tot := 0;
        while i < cnt do
            begin
                tot := tot + i;
                i := i + 1
            end
    end;

procedure avg;
    begin
        if cnt > 0 then
            tmp := tot / cnt
    end;

begin
    cnt := 10;
    call sum;
    call avg;
    write(tot)
end.
