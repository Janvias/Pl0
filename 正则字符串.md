# PL/0 可直接运行C\+\+正则字符串（适配正则自动机模块）

以下所有正则字符串**完全适配前文RegexParser、NFA\_Builder、DFA\_Subset、DFA\_Minimize四大类**，转义合规、可直接传入中缀转后缀函数，用于生成PL/0全套词法、语法自动机。

核心说明：适配自定义正则语法，支持 `|`或、`*`闭包、`+`正闭包、括号分组，自动机模块可直接解析运行。

## 一、PL/0 基础词法单元正则（核心必用）

```cpp
// 1. 标识符：字母开头，后续字母/数字
const string REG_IDENT = "([a-zA-Z][a-zA-Z0-9]*)";

const string REG_NUMBER = "([0-9]+)";
```

## 二、PL/0 语法层核心正则（对应CFG规则）

```cpp
// ==============================================
// 重要：以下正则 100% 兼容自定义 RegexParser
// 不使用 .+? 等不支持惰性匹配、通配符
// 仅支持：() 分组、| 或、* 闭包、+ 正闭包
// 完全适配 NFA_Builder / DFA_Subset / DFA_Minimize
// ==============================================

// 常量定义：标识符=数字
const string REG_CONST_DEF = "([a-zA-Z][a-zA-Z0-9]*=[0-9]+)";

// 常量说明部分：const 常量定义(,常量定义)* ;
const string REG_CONST_PART = "(const[a-zA-Z][a-zA-Z0-9]*=[0-9]+(,[a-zA-Z][a-zA-Z0-9]*=[0-9]+)*;)";

// 变量说明部分：var 标识符(,标识符)* ;
const string REG_VAR_PART = "(var[a-zA-Z][a-zA-Z0-9]*(,[a-zA-Z][a-zA-Z0-9]*)*;)";

// 过程首部：procedure 标识符 ;
const string REG_PROC_HEAD = "(procedure[a-zA-Z][a-zA-Z0-9]*;)";

// 加减运算符、乘除运算符、关系运算符（基础集合）
const string REG_ADD_OP = "(\\+|-)";
const string REG_MUL_OP = "(\\*|/)";
const string REG_REL_OP = "(=|#|<|<=|>|>=)";

// 因子：ID / NUM / (表达式)
const string REG_FACTOR = "([a-zA-Z][a-zA-Z0-9]*|[0-9]+|\\(([a-zA-Z0-9\\+\\-\\*/])*\\))";

// 项：因子 (乘除运算符 因子)*
const string REG_TERM = "(" + REG_FACTOR + "(" + REG_MUL_OP + REG_FACTOR + ")*)";

// 表达式：(+/-)? 项 (加减运算符 项)*
const string REG_EXPR = "(([\\+\\-])?" + REG_TERM + "(" + REG_ADD_OP + REG_TERM + ")*)";

// 条件：表达式 关系符 表达式 | odd 表达式
const string REG_CONDITION = "((" + REG_EXPR + REG_REL_OP + REG_EXPR + ")|(odd" + REG_EXPR + "))";

// 赋值语句：标识符 := 表达式
const string REG_ASSIGN = "([a-zA-Z][a-zA-Z0-9]*:=" + REG_EXPR + ")";

// 条件语句：if 条件 then 语句（语句复用基础单元）
const string REG_IF_STMT = "(if" + REG_CONDITION + "then" + REG_ASSIGN + ")";

// 循环语句：while 条件 do 语句
const string REG_WHILE_STMT = "(while" + REG_CONDITION + "do" + REG_ASSIGN + ")";

// 过程调用语句：call 标识符
const string REG_CALL_STMT = "(call[a-zA-Z][a-zA-Z0-9]*)";

// 读语句：read(标识符(,标识符)*)
const string REG_READ_STMT = "(read\\([a-zA-Z][a-zA-Z0-9]*(,[a-zA-Z][a-zA-Z0-9]*)*\\))";

// 写语句：write(表达式(,表达式)*)
const string REG_WRITE_STMT = "(write\\(" + REG_EXPR + "(," + REG_EXPR + ")*\\))";

// 复合语句：begin 语句(;语句)* end
const string REG_COMPOUND_STMT = "(begin" + REG_ASSIGN + "(;" + REG_ASSIGN + ")*end)";

// PL/0 完整程序框架：常量段可选 + 变量段可选 + 语句体 + 结束符.
const string REG_PL0_PROGRAM = "(" + REG_CONST_PART + "?" + REG_VAR_PART + "?" + REG_COMPOUND_STMT + "\\.)";
```

## 三、通用调用示例（可直接嵌入自动机代码）

```cpp
int main() {
    // 初始化正则解析器 & 自动机
    RegexParser parser;
    NFA_Builder nfaBuilder;
    DFA_Subset dfaBuilder;
    DFA_Minimize dfaMin;

    // 测试用正则列表（PL0核心可解析单元）
    vector<string> testRegs = {
        REG_IDENT,
        REG_NUMBER,
        REG_KEYWORD,
        REG_CONST_DEF,
        REG_VAR_PART,
        REG_EXPR,
        REG_CONDITION,
        REG_ASSIGN,
        REG_COMPOUND_STMT
    };

    // 批量测试：正则 -> 后缀 -> NFA -> DFA -> 最小化DFA
    for (auto& reg : testRegs) {
        cout << "========================" << endl;
        cout << "原正则：" << reg << endl;

        // 1. 中缀转后缀
        string post = parser.infixToPostfix(reg);
        cout << "后缀式：" << post << endl;

        // 2. 生成NFA
        stateCount = 0; // 重置状态计数器
        nfaBuilder.edges.clear();
        nfaBuilder.buildNFA(post);
        cout << "NFA总状态数：" << stateCount << endl;

        // 3. NFA转DFA
        dfaBuilder.buildDFA(nfaBuilder);
        cout << "DFA状态数：" << dfaBuilder.dfaStates.size() << endl;

        // 4. DFA最小化
        dfaMin.minimize(dfaBuilder);
        cout << "最小DFA状态数：" << dfaMin.minDfa.size() << endl;
    }

    cout << "========================" << endl;
    cout << "PL0 全部自动机构造完成！" << endl;

    return 0;
}
```
